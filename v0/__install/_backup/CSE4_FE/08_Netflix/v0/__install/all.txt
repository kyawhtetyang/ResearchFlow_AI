# v0/components/AIChat.tsx

import React, { useState } from 'react';
import { getMovieRecommendations } from '../services/geminiService';

const AIChat: React.FC = () => {
  const [mood, setMood] = useState('');
  const [recommendations, setRecommendations] = useState<any[]>([]);
  const [loading, setLoading] = useState(false);

  const handleSubmit = async (e: React.FormEvent) => {
    e.preventDefault();
    if (!mood.trim()) return;
    setLoading(true);
    try {
      const recs = await getMovieRecommendations(mood);
      setRecommendations(recs);
    } catch (err) {
      console.error(err);
    } finally {
      setLoading(false);
    }
  };

  return (
    <div className="min-h-screen pt-24 pb-12 px-4 md:px-12 max-w-4xl mx-auto">
      <div className="text-center mb-12">
        <h1 className="text-3xl md:text-5xl font-bold mb-4">What should you watch?</h1>
        <p className="text-gray-400 text-lg">Tell Gemini how you're feeling, and we'll craft the perfect watchlist.</p>
      </div>

      <form onSubmit={handleSubmit} className="mb-12 relative">
        <input
          type="text"
          value={mood}
          onChange={(e) => setMood(e.target.value)}
          placeholder="e.g., I want something dark and visually stunning like Bladerunner..."
          className="w-full bg-gray-800/50 border border-gray-700 rounded-xl py-4 px-6 text-xl focus:outline-none focus:ring-2 ring-red-600 transition-all"
        />
        <button
          type="submit"
          disabled={loading}
          className="absolute right-3 top-3 bg-red-600 px-6 py-2 rounded-lg font-bold hover:bg-red-700 disabled:opacity-50"
        >
          {loading ? <i className="fas fa-circle-notch fa-spin"></i> : 'Ask Gemini'}
        </button>
      </form>

      <div className="grid gap-6">
        {recommendations.map((rec, i) => (
          <div key={i} className="bg-gray-800/30 p-6 rounded-xl border border-gray-700 hover:border-red-600/50 transition-all flex flex-col md:flex-row gap-6 animate-fadeIn">

import React, { useRef } from 'react';
import { Movie } from '../types';

interface MovieRowProps {
  title: string;
  movies: Movie[];
}

const MovieRow: React.FC<MovieRowProps> = ({ title, movies }) => {
  const rowRef = useRef<HTMLDivElement>(null);

  const scroll = (direction: 'left' | 'right') => {
    if (rowRef.current) {
      const { scrollLeft, clientWidth } = rowRef.current;
      const scrollTo = direction === 'left' ? scrollLeft - clientWidth : scrollLeft + clientWidth;
      rowRef.current.scrollTo({ left: scrollTo, behavior: 'smooth' });
    }
  };

  return (
    <div className="space-y-2 py-4 group relative">
      <h2 className="text-xl md:text-2xl font-bold px-4 md:px-12 hover:text-white transition-colors cursor-pointer inline-block">

          <i className="fas fa-chevron-left"></i>

# v0/services/geminiService.ts

import { GoogleGenAI, Type, GenerateContentResponse } from "@google/genai";

const getAI = () => new GoogleGenAI({ apiKey: process.env.API_KEY || '' });

export const getMovieRecommendations = async (mood: string) => {
  const ai = getAI();
  const response = await ai.models.generateContent({
    model: 'gemini-3-flash-preview',
    contents: `Recommend 5 movies based on the mood: ${mood}. Provide the response in JSON format.`,
    config: {
      responseMimeType: "application/json",
      responseSchema: {
        type: Type.ARRAY,
        items: {
          type: Type.OBJECT,
          properties: {
            title: { type: Type.STRING },
            reason: { type: Type.STRING },
            year: { type: Type.NUMBER },
            genre: { type: Type.ARRAY, items: { type: Type.STRING } }
          },
          required: ['title', 'reason', 'year', 'genre']
        }
      }
    }
  });
  return JSON.parse(response.text || '[]');
};

export const searchMoviesAI = async (query: string) => {
  const ai = getAI();
  const response = await ai.models.generateContent({
    model: 'gemini-3-flash-preview',
    contents: `Find information about movies or shows related to: ${query}`,
    config: {
      tools: [{ googleSearch: {} }],
    },
  });

  const text = response.text || "No results found.";
  const groundingChunks = response.candidates?.[0]?.groundingMetadata?.groundingChunks || [];
  const sources = groundingChunks
    .filter((c: any) => c.web)
    .map((c: any) => ({
      title: c.web.title,
      url: c.web.uri
    }));

  return { text, sources };
};

export const generateTrailers = async (prompt: string, onStatus: (s: string) => void) => {
  // Check for billing and selected API key
  if (!(await (window as any).aistudio?.hasSelectedApiKey())) {
    await (window as any).aistudio?.openSelectKey();
    // After selection, proceed (handling race condition by just assuming we are good to go)
  }

  const ai = getAI();
  onStatus("Initiating Dream Forge...");

  try {
    let operation = await ai.models.generateVideos({
      model: 'veo-3.1-fast-generate-preview',
      prompt: `A high-quality movie teaser trailer: ${prompt}`,
      config: {
        numberOfVideos: 1,
        resolution: '720p',
        aspectRatio: '16:9'
      }
    });

    onStatus("Gemini is imagining your scene...");

    while (!operation.done) {
      await new Promise(resolve => setTimeout(resolve, 8000));
      onStatus("Synthesizing cinematic frames...");
      operation = await ai.operations.getVideosOperation({ operation: operation });
    }

    const downloadLink = operation.response?.generatedVideos?.[0]?.video?.uri;
    const response = await fetch(`${downloadLink}&key=${process.env.API_KEY}`);
    const blob = await response.blob();
    return URL.createObjectURL(blob);
  } catch (error: any) {
    if (error.message?.includes("Requested entity was not found")) {
      await (window as any).aistudio?.openSelectKey();
    }
    throw error;
  }
};

# v0/App.tsx

import React, { useState, useEffect } from 'react';
import Navbar from './components/Navbar';
import Hero from './components/Hero';
import MovieRow from './components/MovieRow';
import AIChat from './components/AIChat';
import VideoGen from './components/VideoGen';
import { ViewState, Movie } from './types';
import { MOCK_MOVIES, CATEGORIES } from './constants';
import { searchMoviesAI } from './services/geminiService';

const App: React.FC = () => {
  const [view, setView] = useState<ViewState>(ViewState.HOME);
  const [searchQuery, setSearchQuery] = useState('');
  const [searchResults, setSearchResults] = useState<{ text: string, sources: any[] } | null>(null);
  const [searchLoading, setSearchLoading] = useState(false);

  const handleSearch = async (e: React.FormEvent) => {
    e.preventDefault();
    if (!searchQuery.trim()) return;
    setSearchLoading(true);
    try {
      const results = await searchMoviesAI(searchQuery);
      setSearchResults(results);
    } catch (err) {
      console.error(err);
    } finally {
      setSearchLoading(false);
    }
  };

  const renderContent = () => {
    switch (view) {
      case ViewState.AI_CHAT:
        return <AIChat />;
      case ViewState.VIDEO_GEN:
        return <VideoGen />;
      case ViewState.SEARCH:
        return (
          <div className="min-h-screen pt-32 px-4 md:px-12 max-w-5xl mx-auto">
             <form onSubmit={handleSearch} className="mb-8">
                <input
                  type="text"
                  autoFocus
                  value={searchQuery}
                  onChange={(e) => setSearchQuery(e.target.value)}
                  placeholder="Search globally for any movie, show or trivia..."
                  className="w-full bg-transparent border-b-2 border-gray-700 text-4xl md:text-6xl font-bold py-4 focus:outline-none focus:border-red-600 transition-colors"
                />
             </form>

             {searchLoading && (
               <div className="flex flex-col items-center py-20 gap-4">
                  <div className="w-12 h-12 border-4 border-red-600 border-t-transparent rounded-full animate-spin"></div>
                  <p className="text-gray-400 font-mono">Querying Gemini Search...</p>
               </div>
             )}

             {searchResults && !searchLoading && (
               <div className="space-y-8 animate-fadeIn">
                  <div className="prose prose-invert max-w-none text-xl leading-relaxed text-gray-300 whitespace-pre-wrap">
                    {searchResults.text}
                  </div>
                  {searchResults.sources.length > 0 && (
                    <div className="border-t border-gray-800 pt-8">
                       <h3 className="text-sm font-bold text-gray-500 uppercase tracking-widest mb-4">Sources from Web</h3>
                       <div className="grid grid-cols-1 md:grid-cols-2 gap-4">
                          {searchResults.sources.map((src, i) => (
                            <a key={i} href={src.url} target="_blank" rel="noopener noreferrer" className="bg-gray-800/50 p-4 rounded hover:bg-gray-800 transition-colors flex items-center justify-between border border-transparent hover:border-gray-700">
                               <span className="truncate mr-4">{src.title}</span>
                               <i className="fas fa-external-link-alt text-xs text-red-600"></i>
                            </a>
                          ))}
                       </div>
                    </div>
                  )}
               </div>
             )}
          </div>
        );
      case ViewState.HOME:
      default:
        return (
          <main className="pb-20">
            <Hero movie={MOCK_MOVIES[0]} />
            <div className="-mt-16 relative z-30 space-y-8">
              {CATEGORIES.map((cat) => (
                <MovieRow key={cat} title={cat} movies={[...MOCK_MOVIES].sort(() => Math.random() - 0.5)} />
              ))}
            </div>
          </main>
        );
    }
  };

  return (
    <div className="min-h-screen bg-[#141414] text-white">
      <Navbar currentView={view} setView={setView} />
      {renderContent()}

      {/* Floating Action Button for AI Help */}
      {view === ViewState.HOME && (
        <button
          onClick={() => setView(ViewState.AI_CHAT)}
          className="fixed bottom-8 right-8 w-14 h-14 bg-red-600 rounded-full shadow-2xl flex items-center justify-center hover:scale-110 transition-transform z-50 group"
        >
          <i className="fas fa-robot text-2xl"></i>
          <span className="absolute right-full mr-4 bg-white text-black text-sm font-bold py-1 px-3 rounded whitespace-nowrap opacity-0 group-hover:opacity-100 transition-opacity">
            Ask Gemini Recommendation
          </span>
        </button>
      )}

      {/* Footer */}
      <footer className="py-12 px-4 md:px-12 border-t border-gray-800 bg-black/50 text-gray-500 text-sm">
        <div className="max-w-6xl mx-auto flex flex-col md:flex-row justify-between items-center gap-8">
          <div className="flex gap-8">
            <a href="#" className="hover:text-white transition-colors">Audio and Subtitles</a>
            <a href="#" className="hover:text-white transition-colors">Media Center</a>
            <a href="#" className="hover:text-white transition-colors">Privacy</a>

          </div>

  );
};
